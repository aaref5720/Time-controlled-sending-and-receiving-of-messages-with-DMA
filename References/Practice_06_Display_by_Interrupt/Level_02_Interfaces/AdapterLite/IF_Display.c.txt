/*
 * The functionalities in this file belong to the Service Interface Layer of the application.
 * The services build the interface to the functions to operate the 4 segment display on the THD AdapterLite.
 *
 *
 * The Interface shall provide the functions:
 * 1) Func_Adapter_Display_init         initialization of the display interface
 * 2) Func_SetDisplay                   function which takes characters to be displayed in a special format
 * 3) Func_EncodeDisplayInput           function which converts characters to 7-segment format
 * 4) Func_BuildDisplayOutput           function which converts to serial data format
 * 5) Func_Build_ASCLIN_FIFO_Format     function which converts to ASCLIN FIFO format
 * 6) Func_updateDisplay                function which includes the upper 3 functions
 *
 * The display is connected to the TC375 using a SPI - like interface.
 */

/*
 * status: V01.0 2025-11-10
 */

/*
 * the field with the position code of the display elements.
 */
const uint8 DisplayElementPositionCode[] = {0xF1,0xF2,0xF4,0xF8};


/*
 * the field with characters to be displayed.
 * DisplayInput[0] is the character on the most right hand side on the display.
 */
DisplayPrompt DisplayInputBuffer[4];


/*
 * the field with characters to be transferred to the display (encoded for the display).
 * DisplayInput[0] is the character on the most right hand side.
 */
uint8 DisplayInputEncoded[4];


/*
 * the field with the data to be output to the display elements.
 */
DisplayElementStruct DisplayOutputBuffer[4];


/*
 * the field with the data to be output to the display elements, formatted for the ASCLIN FIFO.
 */
uint16 DisplayOutputFIFO[4];

/*

 * returns:     L02_Return_Type Error report
 * parameters:  void
 * description: The function initializes the interface to the display on the MF Shield board.
 */

L02_Return_Type Func_Adapter_Display_init(void)
{

}


/*********************************************************************************************************************/
/*
 * function:    Func_SetDisplayBuffer(DisplayPrompt character, int position)
 * returns:     L02_Return_Type Error report
 * parameters:  DisplayPrompt character: character to be displayed
 *              int position: position of character, allowed range 0..3
 * description: The function stores one character in display input buffer.
 */

L02_Return_Type Func_SetDisplay(DisplayPrompt character, int position)
{
    if((position >= 0) | (position <= 3))
    {
        DisplayInputBuffer[position] = character;
        return L02_E_OK;
    }
    else
    {
        return L02_E_NOT_OK;
    }
}


/*********************************************************************************************************************/
/*
 * function:    Func_EncodeDisplayInput
 * returns:     L02_Return_Type Error report
 * parameters:  void
 * description: The function encodes the values to be displayed to a code as required by the display.
 */

L02_Return_Type Func_EncodeDisplayInput(DisplayPrompt inputfield[], uint8 outputfield[])
{
    int i;

    for (i = 0; i<4; i++)
    {
        switch (inputfield[i])
        {
            case DispPrompt_0noDot:
                outputfield[i] = 0xC0;
                break;
            case DispPrompt_1noDot:
                outputfield[i] = 0xF9;
                break;
            case DispPrompt_2noDot:
                 outputfield[i] = 0xA4;
                 break;
            case DispPrompt_3noDot:
                 outputfield[i] = 0xB0;
                 break;
            case DispPrompt_4noDot:
                outputfield[i] = 0x99;
                break;
            case DispPrompt_5noDot:
                outputfield[i] = 0x92;
                break;
            case DispPrompt_6noDot:
                 outputfield[i] = 0x82;
                 break;
            case DispPrompt_7noDot:
                 outputfield[i] = 0xF8;
                 break;
            case DispPrompt_8noDot:
                 outputfield[i] = 0x80;
                 break;
            case DispPrompt_9noDot:
                 outputfield[i] = 0x90;
                 break;
            case DispPrompt_ONnoDot:
                 outputfield[i] = 0x80;
                 break;
            case DispPrompt_0withDot:
                outputfield[i] = 0x40;
                break;
            case DispPrompt_1withDot:
                outputfield[i] = 0x79;
                break;
            case DispPrompt_2withDot:
                 outputfield[i] = 0x24;
                 break;
            case DispPrompt_3withDot:
                 outputfield[i] = 0x30;
                 break;
            case DispPrompt_4withDot:
                 outputfield[i] = 0x19;
                break;
            case DispPrompt_5withDot:
                 outputfield[i] = 0x12;
                break;
            case DispPrompt_6withDot:
                 outputfield[i] = 0x02;
                 break;
            case DispPrompt_7withDot:
                 outputfield[i] = 0x78;
                 break;
            case DispPrompt_8withDot:
                 outputfield[i] = 0x00;
                 break;
            case DispPrompt_9withDot:
                 outputfield[i] = 0x10;
                 break;
            case DispPrompt_OFF:
                 outputfield[i] = 0xFF;
                 break;
            case DispPrompt_Dot:
                 outputfield[i] = 0x7F;
                 break;
            default:
                outputfield[i] = 0xFF;
        }
    }

    return L02_E_OK;
}


/*********************************************************************************************************************/
/*
 * function:    Func_BuildDisplayOutput
 * returns:     L02_Return_Type Error report
 * parameters:  void
 * description: The function stores the values to be displayed in a format which is required for the serial transfer.
 */

L02_Return_Type Func_BuildDisplayOutput(uint8 inputfield[], DisplayElementStruct outputfield[])
{
    int i;

    for (i = 0; i < 4; i++)
    {
        outputfield[i].ElementShowCode = inputfield[i];
        outputfield[i].ElementPositionCode = DisplayElementPositionCode[i];
    }

    return L02_E_OK;
}


/*********************************************************************************************************************/
/*
 * function:    Func_Build_ASCLIN_FIFO_Format
 * returns:     L02_Return_Type Error report
 * parameters:  void
 * description: The function stores the values to be displayed in a format which is required by the FIFO of ASCLIN.
 */

L02_Return_Type Func_Build_ASCLIN_FIFO_Format(DisplayElementStruct inputfield[], uint16 outputfield[])
{
    int i;

    for (i = 0; i < 4; i++)
    {
        outputfield[i] = 0x0;
        outputfield[i] = ((inputfield[i].ElementShowCode) << 8)|(inputfield[i].ElementPositionCode);
    }

    return L02_E_OK;
}



/*********************************************************************************************************************/
/*
 * function:    Func_updateDisplay(void)
 * returns:     L02_Return_Type Error report
 * parameters:  void
 * description: The function converts the input characters into display characters and outputs them on the display.
 */

L02_Return_Type Func_updateDisplay(void)
{
    L02_Return_Type tmp_L02;
    // HAL_Return_Type tmp_HAL;

    tmp_L02 = Func_EncodeDisplayInput(DisplayInputBuffer, DisplayInputEncoded);
    tmp_L02 = Func_BuildDisplayOutput(DisplayInputEncoded, DisplayOutputBuffer);
    tmp_L02 = Func_Build_ASCLIN_FIFO_Format(DisplayOutputBuffer, DisplayOutputFIFO);

    return L02_E_OK;
}
