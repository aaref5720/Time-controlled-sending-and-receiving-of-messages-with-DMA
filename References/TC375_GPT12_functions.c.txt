/*
 * The code in this file shall provide the following functionalities:
 *
 * 1) initialize the GPT12 timer
 * 2) start the GPT12 timer
 * 3) stop the GPT12 timer
 * 4) (re)-load the GPT12 timer values
 */

/*********************************************************************************************************************/
/*-----------------------------------------------------Defines-------------------------------------------------------*/
/*********************************************************************************************************************/

/* * Frequency Calculation for T6:
 * fGPT = 100MHz.
 * If BPS2 = 1 (fGPT/2) and T6I = 7 (fGPT/128), total prescaler is 256.
 * fT6 = 390,625 Hz.
 * For 1s: 390,625 ticks (Exceeds 16-bit 0xFFFF).
 * Note: You may need to increase BPS2 or T6I further for exact 1s/2s timing.
 */
#define T6_RELOAD_VAL_1S 0x4144 /* ~500ms with prescaler 1024 */
#define T6_RELOAD_VAL_2S 0x080C /* ~650ms with prescaler 1024 */

/*
 * function:    Func_GPT12_Module_Init
 * returns:     void
 * parameters:  void
 * description: The function enables the GPT12 module and switches on the module
 * clock.
 */
void Func_GPT12_Module_Init(void) {
  Func_clear_Global_ENDINIT();

  /* Enable GPT12 module clock */
  GPT120_CLC.B.DISR = 0x0;
  GPT120_CLC.B.EDIS = 0x1;

  /* Dummy read to ensure write is finished */
  volatile uint32 tmp_read = GPT120_CLC.U;
  (void)tmp_read;

  Func_set_Global_ENDINIT();
}

/*********************************************************************************************************************/
/*
 * function:    Func_GPT12_T6_init
 * returns:     HAL_Return_Type error report
 * parameters:  void
 * description: The function initializes the GPT12 module for operation
 */
HAL_Return_Type Func_GPT12_T6_init(void) {
  HAL_Return_Type tmp_HAL = HAL_E_NOT_OK;

  if (GPT120_CLC.B.DISS == 0x0) // DISS == 0x0 -> GPT12 is successfully enabled
  {
    /*
     * the timer register T6 is configured for timer purposes
     *
     * field 02..00: T6I: 0x3, T6 internal divisor is set to 3
     * field 05..03: T6M: 0x0, T6 is set to timer mode
     * field 06: T6R: 0x0, T6 is stopped (it will be started separately)
     * field 07: T6UD: 0x1, T6 counts down
     * field 08: T6UDE: 0x0, count direction is controlled by T6UD
     * field 09: T6OE: 0x0, alternate output function disabled
     * field 10: T6OTL: 0x0, not used
     * field 12..11: BPS2: 0x2, divisor is - in combination with T6I - 128
     * field 14: T6CLR: 0x0, T6 is not cleared on a capture event
     * field 15: T6SR: 0x1, reload from CAPREL enabled
     */

    GPT120_T6CON.B.T6M = 0x0;
    GPT120_T6CON.B.T6I = 0x3;
    GPT120_T6CON.B.BPS2 = 0x2;
    GPT120_T6CON.B.T6UD = 0x1;
    GPT120_T6CON.B.T6UDE = 0x0;
    GPT120_T6CON.B.T6CLR = 0x0;
    GPT120_T6CON.B.T6SR = 0x1;
    GPT120_T6CON.B.T6R = 0x0;
    GPT120_T6CON.B.T6OE = 0x0;

    tmp_HAL = HAL_E_OK;
  } else {
    tmp_HAL = HAL_E_NOT_OK;
  }

  return tmp_HAL;
}

/*********************************************************************************************************************/
/*
 * function:    Func_GPT12_T6_start
 * returns:     HAL_Return_Type Error report
 * parameters:  void
 * description: The function starts the timer T6 of GPT12
 */
HAL_Return_Type Func_GPT12_T6_start(void) {
  GPT120_T6CON.B.T6R = 0x1; // T6 started
  return HAL_E_OK;
}

/*********************************************************************************************************************/
/*
 * function:    Func_GPT12_T6_stop
 * returns:     HAL_Return_Type Error report
 * parameters:  void
 * description: The function stops the timer T6 of GPT12
 */
HAL_Return_Type Func_GPT12_T6_stop(void) {
  GPT120_T6CON.B.T6R = 0x0; // T6 stopped
  return HAL_E_OK;
}

/*********************************************************************************************************************/
/*
 * function:    Func_GPT12_T6_set_value
 * returns:     HAL_Return_Type Error report
 * parameters:  uint16 timer_value: value to be loaded into T6
 * description: The function sets the counter value of timer T6 of GPT12
 */
HAL_Return_Type Func_GPT12_T6_set_value(uint16 timer_value) {
  GPT120_T6.U = (uint32)timer_value;
  return HAL_E_OK;
}

/*********************************************************************************************************************/
/*
 * function:    Func_GPT12_CAPREL_set_value
 * returns:     HAL_Return_Type Error report
 * parameters:  uint16 timer_value: value to be loaded into CAPREL
 * description: The function sets the counter value of timer CAPREL of GPT12
 */
HAL_Return_Type Func_GPT12_CAPREL_set_value(uint16 timer_value) {
  GPT120_CAPREL.U = (uint32)timer_value;
  return HAL_E_OK;
}

/*********************************************************************************************************************/
/*
 * function:    Func_GPT12_T3_Init_Counter
 * returns:     void
 * parameters:  void
 * description: The function initializes the GPT12 Timer T3 as a counter
 */
void Func_GPT12_T3_Init_Counter(void) {
  /*
   * Configure T3 as a counter
   * T3M = 0x0: Timer mode
   * T3I = 0x0: Input clock selection
   * T3R = 0x0: Timer stopped initially
   * T3UD = 0x0: Count up
   * T3UDE = 0x0: External direction control disabled
   * T3OE = 0x0: Alternate output function disabled
   */
  GPT120_T3CON.B.T3M = 0x0;
  GPT120_T3CON.B.T3I = 0x0;
  GPT120_T3CON.B.T3R = 0x1; // Start the counter
  GPT120_T3CON.B.T3UD = 0x0;
  GPT120_T3CON.B.T3UDE = 0x0;
  GPT120_T3CON.B.T3OE = 0x0;

  /* Initialize T3 value to 0 */
  GPT120_T3.U = 0x0;
}

/*********************************************************************************************************************/
/*
 * function:    Func_GPT12_T6_Init_Timer
 * returns:     void
 * parameters:  void
 * description: The function initializes the GPT12 Timer T6 and starts it
 */
void Func_GPT12_T6_Init_Timer(void) {
  /* Initialize T6 using the existing init function */
  Func_GPT12_T6_init();

  /* Set initial period to 1s */
  Func_GPT12_CAPREL_set_value(T6_RELOAD_VAL_1S);
  Func_GPT12_T6_set_value(T6_RELOAD_VAL_1S);

  /* Start T6 */
  Func_GPT12_T6_start();
}

/*********************************************************************************************************************/
/*
 * function:    Func_GPT12_T6_Set_Period_1s
 * returns:     void
 * parameters:  void
 * description: The function sets the GPT12 Timer T6 period to approximately 1
 * second
 */
void Func_GPT12_T6_Set_Period_1s(void) {
  Func_GPT12_CAPREL_set_value(T6_RELOAD_VAL_1S);
  Func_GPT12_T6_set_value(T6_RELOAD_VAL_1S);
}

/*********************************************************************************************************************/
/*
 * function:    Func_GPT12_T6_Set_Period_2s
 * returns:     void
 * parameters:  void
 * description: The function sets the GPT12 Timer T6 period to approximately 2
 * seconds
 */
void Func_GPT12_T6_Set_Period_2s(void) {
  Func_GPT12_CAPREL_set_value(T6_RELOAD_VAL_2S);
  Func_GPT12_T6_set_value(T6_RELOAD_VAL_2S);
}
